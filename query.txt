DATE_FORMAT(time_reports.date, '%m/%d/%Y') AS date, WEEKDAY(date) AS weekday

DATE_FORMAT(time_reports.date, '%m/%d/%Y') AS date, DAYNAME(date) AS weekday

SELECT time_reports.id, time_reports.employee_id, employees.name, ROUND(time_reports.hours, 2) AS hours,
DATE_FORMAT(time_reports.date, '%m/%d/%Y') AS date, DAYNAME(date) AS weekday
FROM time_reports INNER JOIN employees ON employees.id = time_reports.employee_id ORDER BY date;
